Cliente ${data.razon}
Rango ${data.rango}
Saldo Disponible ${moneda.format(data.saldo)}
`; } catch (error) { console.error("Error consultando cliente:", error); setLoadingState(false); mostrarError("Error de conexión con el servidor."); } } /* ========================= ROBAPÁGINAS Controla la rotación de banners laterales ========================= */ function mostrarBanner(imgElementId, imageArray, index) { const img = document.getElementById(imgElementId); if (!img || !imageArray.length) return; img.classList.remove("visible"); setTimeout(() => { img.src = imageArray[index]; img.classList.add("visible"); }, 250); } function iniciarRotacionBanners() { if (window.innerWidth <= 1200) return; if (!webappConfig.leftImages.length && !webappConfig.rightImages.length) return; if (webappConfig.leftImages.length) { mostrarBanner("left-banner", webappConfig.leftImages, leftIndex); setInterval(() => { leftIndex = (leftIndex + 1) % webappConfig.leftImages.length; mostrarBanner("left-banner", webappConfig.leftImages, leftIndex); }, 6000); } if (webappConfig.rightImages.length) { mostrarBanner("right-banner", webappConfig.rightImages, rightIndex); setInterval(() => { rightIndex = (rightIndex + 1) % webappConfig.rightImages.length; mostrarBanner("right-banner", webappConfig.rightImages, rightIndex); }, 6000); } } /* ========================= EVENTOS DE TECLADO Y ARRANQUE Carga config al abrir y permite consultar con Enter ========================= */ document.addEventListener("DOMContentLoaded", function () { const nitInput = document.getElementById("nit-in"); const passInput = document.getElementById("pass-in"); function manejarEnter(e) { if (e.key === "Enter") { e.preventDefault(); ejecutar(); } } nitInput.addEventListener("keydown", manejarEnter); passInput.addEventListener("keydown", manejarEnter); cargarConfig(); });
